36 /// The base class for probabilistic events in a fault tree.
37 ///
38 /// The base event item provides
39 /// only the boxes containing the name and description of the event.
40 /// A derived class must provide
41 /// the symbolic representation of its kind.
42 ///
43 /// The sizes are measured in units of character height and average width.
44 /// This class provides the reference units for derived classes to use.
45 /// All derived class shapes should stay within the allowed box limits
46 /// to make the fault tree structure layered.
47 class Event : public QGraphicsItem
48 {
49 public:
50  ~Event() noexcept override;
51 
52  /// Required standard member functions of QGraphicsItem interface.
53  /// @{
54  QRectF boundingRect() const final;
55  void paint(QPainter *painter, const QStyleOptionGraphicsItem *option,
56  QWidget *widget) final;
57  /// @}
58 
59  /// @returns The width of the whole subgraph.
60  virtual double width() const;
61 
62  /// @returns The model data event.
63  model::Element *data() const { return m_event; }
64 
65 protected:
66  /// The confining size of the Event graphics in characters.
67  /// The derived event types should stay within this confinement.
68  static const QSizeF m_size;
69  /// The height of the confining space used only by the Event base class.
70  static const double m_baseHeight;
71  /// The length of the ID box in characters.
72  /// The height of the ID box is 1 character.
73  static const double m_idBoxLength;
74  /// The height of the Label box in characters.
75  static const double m_labelBoxHeight;
76 
77  /// Assigns an event to a presentation view.
78  ///
79  /// @param event The model event.
80  /// @param parent The parent Graphics event.
81  explicit Event(model::Element *event, QGraphicsItem *parent = nullptr);
82 
83  /// @returns The graphics of the derived class.
84  QGraphicsItem *getTypeGraphics() const { return m_typeGraphics; }
85 
86  /// Releases the current derived class item, and sets the new one.
87  ///
88  /// @param item The new item to represent the derived type.
89  void setTypeGraphics(QGraphicsItem *item);
90 
91  /// @returns Unit width (x) and height (y) for shapes.
92  QSizeF units() const;
93 
94  model::Element *m_event; ///< The model data.
95 
96 private:
97  QGraphicsItem *m_typeGraphics; ///< The graphics of the derived type.
98  QMetaObject::Connection m_labelConnection; ///< Tracks the label changes.
99  QMetaObject::Connection m_idConnection; ///< Tracks the ID changes.
100 };
