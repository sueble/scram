75 QVariant ImportanceTableModel::data(const QModelIndex &index, int role) const
76 {
77  if (!index.isValid())
78  return {};
79  if (role == Qt::TextAlignmentRole && index.column() != 0)
80  return ALIGN_NUMBER_IN_TABLE;

84  GUI_ASSERT(index.row() < m_data.size(), {});

86  const core::ImportanceRecord &record = m_data[index.row()];
