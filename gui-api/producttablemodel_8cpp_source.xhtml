29 ProductTableModel::ProductTableModel(const core::ProductContainer &products,
30  bool withProbability, QObject *parent)
31  : QAbstractTableModel(parent), m_withProbability(withProbability)
32 {
33  m_products.reserve(products.size());
34  double sum = 0;
35  for (const core::Product &product : products) {
36  QString members;
37  for (auto it = product.begin(), it_end = product.end(); it != it_end;) {
38  const core::Literal &literal = *it;
39  if (literal.complement)
40  members.append(QStringLiteral("\u00AC"));
41  members.append(QString::fromStdString(literal.event.id()));
42  if (++it != it_end)
43  members.append(QStringLiteral(" \u22C5 "));
44  }
45  double probability = m_withProbability ? product.p() : 0;
46  sum += probability;
47  m_products.push_back(
48  {std::move(members), product.order(), probability});
49  }
50 
51  if (sum == 0)
52  return;
53 
54  for (Product &product : m_products)
55  product.contribution = product.probability / sum;
56 }

89 QVariant ProductTableModel::data(const QModelIndex &index, int role) const
90 {
91  if (!index.isValid())
92  return {};
93  if (role == Qt::TextAlignmentRole && index.column() != 0)
94  return ALIGN_NUMBER_IN_TABLE;
95  if (role != Qt::DisplayRole)
96  return {};
97 
98  GUI_ASSERT(index.row() < m_products.size(), {});
99  const Product &product = m_products[index.row()];
100  switch (index.column()) {
101  case 0:
102  return product.toString;
103  case 1:
104  return product.order;
105  case 2:

107  case 3:
108  return product.contribution;
109  }
