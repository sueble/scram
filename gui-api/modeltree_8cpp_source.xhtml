27 ModelTree::ModelTree(model::Model *model, QObject *parent)
28  : QAbstractItemModel(parent), m_model(model)
29 {
30  for (const mef::FaultTreePtr &faultTree : m_model->faultTrees())
31  m_faultTrees.insert(faultTree.get());
32 
33  connect(m_model, OVERLOAD(model::Model, added, mef::FaultTree *), this,
34  [this](mef::FaultTree *faultTree) {
35  auto it = m_faultTrees.lower_bound(faultTree);
36  int index = m_faultTrees.index_of(it);
37  beginInsertRows(
38  this->index(static_cast<int>(Row::FaultTrees), 0, {}),
39  index, index);
40  m_faultTrees.insert(it, faultTree);
41  endInsertRows();
42  });
43  connect(m_model, OVERLOAD(model::Model, removed, mef::FaultTree *), this,
44  [this](mef::FaultTree *faultTree) {
45  auto it = m_faultTrees.find(faultTree);
46  GUI_ASSERT(it != m_faultTrees.end(), );
47  int index = m_faultTrees.index_of(it);
48  beginRemoveRows(
49  this->index(static_cast<int>(Row::FaultTrees), 0, {}),
50  index, index);
51  m_faultTrees.erase(it);
52  endRemoveRows();
53  });
54 
55  setupElementCountConnections<mef::FaultTree, Row::FaultTrees>();
56  setupElementCountConnections<model::Gate, Row::Gates>();
57  setupElementCountConnections<model::BasicEvent, Row::BasicEvents>();
58  setupElementCountConnections<model::HouseEvent, Row::HouseEvents>();
59 }

112 QVariant ModelTree::data(const QModelIndex &index, int role) const
113 {
114  if (!index.isValid())
115  return {};
116  if (role == Qt::UserRole)
117  return QVariant::fromValue(index.internalPointer());
118  if (role != Qt::DisplayRole)
119  return {};
120 
121  if (index.parent().isValid())
122  return QString::fromStdString(
123  static_cast<mef::FaultTree *>(index.internalPointer())->name());
124  switch (static_cast<Row>(index.row())) {
125  case Row::FaultTrees:
126  //: The parent item for collections of fault trees in the model.
127  return tr("Fault Trees (%L1)").arg(m_model->faultTrees().size());
128  case Row::Gates:
129  //: The table of gates.
130  return tr("Gates (%L1)").arg(m_model->gates().size());
131  case Row::BasicEvents:
132  //: The table of basic events.
133  return tr("Basic Events (%L1)").arg(m_model->basicEvents().size());
134  case Row::HouseEvents:
135  //: The table of house events.
136  return tr("House Events (%L1)").arg(m_model->houseEvents().size());
137  }
138  GUI_ASSERT(false, {});
139 }
