27 /// Iterator adaptor for indication of container ``find`` call results.
28 /// Conveniently wraps common calls after ``find`` into implicit Boolean value.
29 ///
30 /// @tparam Iterator Iterator type belonging to the container.
31 template <class Iterator>
32 class find_iterator : public Iterator {
33  public:
34  /// Initializes the iterator as the result of ``find()``.
35  ///
36  /// @param[in] it The result of ``find`` call.
37  /// @param[in] it_end The sentinel iterator indicator ``not-found``.
38  find_iterator(Iterator&& it, const Iterator& it_end)
39  : Iterator(std::move(it)), found_(*this != it_end) {}
40 
41  /// @returns true if the iterator indicates that the item is found.
42  explicit operator bool() { return found_; }
43 
44  private:
45  bool found_; ///< Indicator of the lookup result.
46 };

48 /// Wraps ``container::find()`` calls for convenient and efficient lookup
49 /// with ``find_iterator`` adaptor.
50 ///
51 /// @tparam T Container type supporting ``find()`` and ``end()`` calls.
52 /// @tparam Arg The argument type to the ``find()`` call.
53 ///
54 /// @param[in] container The container to operate upon.
55 /// @param[in] arg The argument to the ``find()`` function.
56 ///
57 /// @returns find_iterator wrapping the resultant iterator.
58 template <class T, typename Arg>
59 auto find(T&& container, Arg&& arg) {
60  auto it = container.find(std::forward<Arg>(arg));
61  return find_iterator<decltype(it)>(std::move(it), container.end());
62 }
