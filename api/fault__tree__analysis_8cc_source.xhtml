34 void Print(const ProductContainer& products) {
35  if (products.empty()) {
36  std::cerr << "No products!" << std::endl;
37  return;
38  }
39  if (products.begin()->empty()) {
40  assert(products.size() == 1 && "Unity case must have only one product.");
41  std::cerr << "Single Unity product." << std::endl;
42  return;
43  }
44  std::cerr << " " << products.size() << " : {";
45  for (int i : products.Distribution())
46  std::cerr << " " << i;
47  std::cerr << " }\n\n";
48 
49  using ProductSet = boost::container::flat_set<std::string>;
50  std::vector<ProductSet> to_print;
51  for (const Product& product : products) {
52  ProductSet ids;
53  for (const Literal& literal : product) {
54  ids.insert((literal.complement ? "~" : "") + literal.event.name());
55  }
56  to_print.push_back(std::move(ids));
57  }
58  boost::sort(to_print, [](const ProductSet& lhs, const ProductSet& rhs) {
59  if (lhs.size() == rhs.size())
60  return lhs < rhs;
61  return lhs.size() < rhs.size();
62  });
63  assert(!to_print.front().empty() && "Failure of the analysis with Unity!");
64 
65  for (const auto& product : to_print) {
66  for (const auto& id : product)
67  std::cerr << " " << id;
68  std::cerr << "\n";
69  }
70  std::cerr << std::flush;
71 }

97 void FaultTreeAnalysis::Analyze() noexcept {
98  CLOCK(analysis_time);
99  graph_ = std::make_unique<Pdag>(top_event_,
100  Analysis::settings().ccf_analysis(), model_);
101  this->Preprocess(graph_.get());
102 #ifndef NDEBUG
103  if (Analysis::settings().preprocessor)
104  return; // Preprocessor only option.
105 #endif
106  CLOCK(algo_time);
107  LOG(DEBUG2) << "Launching the algorithm...";
108  const Zbdd& products = this->GenerateProducts(graph_.get());
109  LOG(DEBUG2) << "The algorithm finished in " << DUR(algo_time);
110  LOG(DEBUG2) << "# of products: " << products.size();
111 
112  Analysis::AddAnalysisTime(DUR(analysis_time));
113  CLOCK(store_time);
114  Store(products, *graph_);
115  LOG(DEBUG2) << "Stored the result for reporting in " << DUR(store_time);
116 }

118 void FaultTreeAnalysis::Store(const Zbdd& products,
119  const Pdag& graph) noexcept {
120  // Special cases of sets.
121  if (products.empty()) {
122  Analysis::AddWarning("The set is NULL/Empty.");
123  } else if (products.base()) {
124  Analysis::AddWarning("The set is UNITY/Base.");
125  }
126  products_ = std::make_unique<const ProductContainer>(products, graph);
127 
128 #ifndef NDEBUG
129  for (const Product& product : *products_)
130  assert(product.size() <= Analysis::settings().limit_order() &&
131  "Miscalculated product sets with larger-than-required order.");
132 
133  if (Analysis::settings().print)
134  Print(*products_);
135 #endif
136 }
