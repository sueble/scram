63  /// Computes the switch-case expression with the given evaluator.
64  template <typename F>
65  double Compute(F&& eval) noexcept {
66  for (Case& case_arm : cases_) {
67  if (eval(&case_arm.condition))
68  return eval(&case_arm.value);
69  }
70  return eval(&default_value_);
71  }
72 
73  private:
74  std::vector<Case> cases_; ///< Ordered collection of cases.
75  Expression& default_value_; ///< The default case value.
