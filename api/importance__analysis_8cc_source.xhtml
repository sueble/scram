35 void ImportanceAnalysis::Analyze() noexcept {
36  CLOCK(imp_time);
37  LOG(DEBUG3) << "Calculating importance factors...";
38  double p_total = this->p_total();
39  const std::vector<const mef::BasicEvent*>& basic_events =
40  this->basic_events();
41 
42  std::vector<int> occurrences = this->occurrences();
43  for (int i = 0; i < basic_events.size(); ++i) {
44  if (occurrences[i] == 0)
45  continue;
46  const mef::BasicEvent& event = *basic_events[i];
47  double p_var = event.p();
48  ImportanceFactors imp{};
49  imp.occurrence = occurrences[i];
50  imp.mif = this->CalculateMif(i);
51  if (p_total != 0) {
52  imp.cif = p_var * imp.mif / p_total;
53  imp.raw = 1 + (1 - p_var) * imp.mif / p_total;
54  imp.dif = p_var * imp.raw;
55  if (p_total != p_var * imp.mif)
56  imp.rrw = p_total / (p_total - p_var * imp.mif);
57  }
58  importance_.push_back({event, imp});
59  }
60  LOG(DEBUG3) << "Calculated importance factors in " << DUR(imp_time);
61  Analysis::AddAnalysisTime(DUR(imp_time));
62 }

86 double ImportanceAnalyzer<Bdd>::CalculateMif(const Bdd::VertexPtr& vertex,
87  int order, bool mark) noexcept {
88  if (vertex->terminal())
89  return 0;
90  Ite& ite = Ite::Ref(vertex);
91  if (ite.mark() == mark)
92  return ite.factor();
93  ite.mark(mark);
94  if (ite.order() > order) {
95  if (!ite.module()) {
96  ite.factor(0);
97  } else { /// @todo Detect if the variable is in the module.
98  // The assumption is
99  // that the order of a module is always larger
100  // than the order of its variables.
101  double high = RetrieveProbability(ite.high());
102  double low = RetrieveProbability(ite.low());
103  if (ite.complement_edge())
104  low = 1 - low;
105  const Bdd::Function& res =
106  bdd_graph_->modules().find(ite.index())->second;
107  double mif = CalculateMif(res.vertex, order, mark);
108  if (res.complement)
109  mif = -mif;
110  ite.factor((high - low) * mif);
111  }
112  } else if (ite.order() == order) {
113  assert(!ite.module() && "A variable can't be a module.");
114  double high = RetrieveProbability(ite.high());
115  double low = RetrieveProbability(ite.low());
116  if (ite.complement_edge())
117  low = 1 - low;
118  ite.factor(high - low);
119  } else {
120  assert(ite.order() < order);
121  double p_var = 0;
122  if (ite.module()) {
123  const Bdd::Function& res =
124  bdd_graph_->modules().find(ite.index())->second;
125  p_var = RetrieveProbability(res.vertex);
126  if (res.complement)
127  p_var = 1 - p_var;
128  } else {
129  p_var = prob_analyzer()->p_vars()[ite.index()];
130  }
131  double high = CalculateMif(ite.high(), order, mark);
132  double low = CalculateMif(ite.low(), order, mark);
133  if (ite.complement_edge())
134  low = -low;
135  ite.factor(p_var * high + (1 - p_var) * low);
136  }
137  return ite.factor();
138 }
