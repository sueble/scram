187  /// @tparam Algorithm Qualitative analysis algorithm.

192  mef::MissionTime* mission_time)

194  graph_(fta->graph()),
195  products_(fta->algorithm()->products()) {
196  ExtractVariableProbabilities();
197  }
198 
199  /// @returns The original PDAG from the fault tree analyzer.
200  const Pdag* graph() const { return graph_; }
201 
202  /// @returns The resulting products of the fault tree analyzer.
203  const Zbdd& products() const { return products_; }
204 
205  /// @returns A mapping for probability values with indices.
206  const Pdag::IndexMap<double>& p_vars() const { return p_vars_; }
207 
208  protected:
209  ~ProbabilityAnalyzerBase() override = default;
210 
211  private:
212  /// Calculates the total probability
213  /// with a different set of probability values
214  /// than the one given upon construction.
215  ///
216  /// @param[in] p_vars A map of probabilities of the graph variables.
217  /// The indices of the variables must map
218  /// exactly to the values.
219  ///
220  /// @returns The total probability calculated with the given values.
221  virtual double
222  CalculateTotalProbability(const Pdag::IndexMap<double>& p_vars) noexcept = 0;
223 
224  double CalculateTotalProbability() noexcept final {
225  return this->CalculateTotalProbability(p_vars_);
226  }
227 
228  std::vector<std::pair<double, double>>
229  CalculateProbabilityOverTime() noexcept final;
230 
231  /// Upon construction of the probability analysis,
232  /// stores the variable probabilities in a continuous container
233  /// for retrieval by their indices instead of pointers.
234  ///
235  /// @note This function may seem redundant,
236  /// for it's super-short and simple to do it inline in the constructor.
237  /// The main benefit of the out-of-line implementation
238  /// is compile-time decoupling from the input BasicEvent classes.
239  void ExtractVariableProbabilities();
240 
241  const Pdag* graph_; ///< PDAG from the fault tree analysis.
242  const Zbdd& products_; ///< A collection of products.
243  Pdag::IndexMap<double> p_vars_; ///< Variable probabilities.
244 };
