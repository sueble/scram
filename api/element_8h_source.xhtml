259 /// Adds a unique element into a table,
260 /// ensuring no duplicated entries.
261 ///
262 /// @tparam ErrorType The error type to indicate the insertion of a duplicate.
263 /// @tparam T A pointer type of the element supporting Element API.
264 /// @tparam Table A set container supporting standard insert API.
265 ///
266 /// @param[in] element The pointer to the unique element.
267 /// @param[in,out] table The destination set container.
268 /// @param[in] header The error message header appearing before duplicate name.
269 ///
270 /// @throws ErrorType The element is already in the table.
271 template <class ErrorType, class T, class Table>
272 void AddElement(T&& element, Table* table, const char* header) {
273  const std::string& name = element->name(); // The element pointer may move!
274  if (table->insert(std::forward<T>(element)).second == false) {
275  throw ErrorType(header + name);
276  }
277 }
