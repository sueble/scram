33 /// Read forward-iterator for combination generation.
34 /// The combination generator guarantees the element order
35 /// to be the same as in the original source collection.
36 ///
37 /// @tparam Iterator A random access iterator type.
38 /// @tparam value_type The container type to generate combinations into.
39 template <class Iterator,
40  class value_type = std::vector<typename Iterator::value_type>>
41 class combination_iterator
42  : public boost::iterator_facade<combination_iterator<Iterator>, value_type,
43  boost::forward_traversal_tag, value_type> {
44  friend class boost::iterator_core_access;
45 
46  public:
47  /// Constructor for a range with N elements to choose from.
48  ///
49  /// @param[in] k The number of elements to choose.
50  /// @param[in] first1 The start of the range.
51  /// @param[in] last1 The sentinel end of the range.
52  combination_iterator(int k, Iterator first1, Iterator last1)
53  : first1_(first1), bitmask_(std::distance(first1, last1)) {
54  assert(k > 0 && "The choice must be positive.");
55  assert(k <= std::distance(first1, last1) && "The choice can't exceed N.");
56  std::fill_n(bitmask_.begin(), k, true);
57  }
58 
59  /// Constructs a special iterator to signal the end of generation.
60  explicit combination_iterator(Iterator first1) : first1_(first1) {}
61 
62  private:
63  /// Standard iterator functionality required by the facade facilities.
64  /// @{
65  void increment() {
66  if (boost::prev_permutation(bitmask_) == false)
67  bitmask_.clear();
68  }
69  bool equal(const combination_iterator& other) const {
70  return first1_ == other.first1_ && bitmask_ == other.bitmask_;
71  }
72  value_type dereference() const {
73  assert(!bitmask_.empty() && "Calling on the sentinel iterator.");
74  value_type combination;
75  for (int i = 0; i < bitmask_.size(); ++i) {
76  if (bitmask_[i])
77  combination.push_back(*std::next(first1_, i));
78  }
79  return combination;
80  }
81  /// @}
82  Iterator first1_; ///< The first element in the collection.
83  std::vector<bool> bitmask_; ///< bit-mask for N elements.
84 };
