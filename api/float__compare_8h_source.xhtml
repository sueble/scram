28 /// Compares two floating point number within tolerance.
29 ///
30 /// @param[in] lhs The reference comparison value.
31 /// @param[in] rhs The test value.
32 /// @param[in] tolerance The percent tolerance for the difference.
33 ///
34 /// @returns true if the values are close within the tolerance.
35 ///
36 /// @note The implementation does not deal with overflows and underflows.
37 /// Therefore, the compared numbers must not be too large or small.
38 inline bool is_close(double lhs, double rhs, double tolerance) noexcept {
39  assert(tolerance < 1 && "Invalid tolerance for float comparison.");
40  return std::abs(lhs - rhs) <= std::abs(lhs) * tolerance;
41 }
