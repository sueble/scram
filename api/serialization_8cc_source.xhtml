38 void Serialize(const Model& model, const std::string& file) {
39  std::unique_ptr<std::FILE, decltype(&std::fclose)> fp(
40  std::fopen(file.c_str(), "w"), &std::fclose);
41  try {
42  if (!fp) {
43  SCRAM_THROW(IOError("Cannot open the output file for serialization."))
44  << boost::errinfo_errno(errno) << boost::errinfo_file_open_mode("w");
45  }
46  Serialize(model, fp.get());
47  } catch (IOError& err) {
48  err << boost::errinfo_file_name(file);
49  throw;
50  }
51 }

55 void SerializeLabelAndAttributes(const Element& element,
56  xml::StreamElement* xml_element) {
57  if (element.label().empty() == false)
58  xml_element->AddChild("label").AddText(element.label());
59  if (element.attributes().empty() == false) {
60  xml::StreamElement attributes_container =
61  xml_element->AddChild("attributes");
62  for (const Attribute& attribute : element.attributes()) {
63  xml::StreamElement attribute_element =
64  attributes_container.AddChild("attribute");
65  assert(attribute.name.empty() == false);
66  attribute_element.SetAttribute("name", attribute.name);
67  assert(attribute.value.empty() == false);
68  attribute_element.SetAttribute("value", attribute.value);
69  if (attribute.type.empty() == false)
70  attribute_element.SetAttribute("type", attribute.type);
71  }
72  }
73 }

80 void Serialize(const Formula& formula, xml::StreamElement* parent) {
81  assert(formula.formula_args().empty());
82  struct ArgStreamer {
83  void operator()(const Gate* gate) const {
84  xml->AddChild("gate").SetAttribute("name", gate->name());
85  }
86  void operator()(const BasicEvent* basic_event) const {
87  xml->AddChild("basic-event").SetAttribute("name", basic_event->name());
88  }
89  void operator()(const HouseEvent* house_event) const {
90  xml->AddChild("house-event").SetAttribute("name", house_event->name());
91  }
92 
93  xml::StreamElement* xml;
94  };
95  if (formula.type() == kNull) {
96  assert(formula.event_args().size() == 1);
97  std::visit(ArgStreamer{parent}, formula.event_args().front());
98  } else {
99  xml::StreamElement type_element = [&formula, &parent] {
100  switch (formula.type()) {
101  case kNot:
102  return parent->AddChild("not");
103  case kAnd:
104  return parent->AddChild("and");
105  case kOr:
106  return parent->AddChild("or");
107  case kNand:
108  return parent->AddChild("nand");
109  case kNor:
110  return parent->AddChild("nor");
111  case kXor:
112  return parent->AddChild("xor");
113  case kVote:
114  return [&formula, &parent] {
115  xml::StreamElement atleast = parent->AddChild("atleast");
116  atleast.SetAttribute("min", formula.vote_number());
117  return atleast;
118  }(); // Wrap NRVO into RVO for GCC.
119  default:
120  assert(false && "Unexpected formula");
121  }
122  }();
123  for (const Formula::EventArg& arg : formula.event_args())
124  std::visit(ArgStreamer{&type_element}, arg);
125  }
126 }

144 void Serialize(const Expression& expression, xml::StreamElement* parent) {
145  if (const auto* constant =
146  dynamic_cast<const ConstantExpression*>(&expression)) {
147  /// @todo Track the original value type of the constant expression.
148  parent->AddChild("float").SetAttribute(
149  "value", const_cast<ConstantExpression*>(constant)->value());
150  } else if (const auto* exponential =
151  dynamic_cast<const Exponential*>(&expression)) {
152  xml::StreamElement xml = parent->AddChild("exponential");
153  assert(exponential->args().size() == 2);
154  for (const Expression* arg : exponential->args())
155  Serialize(*arg, &xml);
156  } else {
157  assert(false && "Unsupported expression");
158  }
159 }

169 void Serialize(const HouseEvent& house_event, xml::StreamElement* parent) {
170  assert(house_event.role() == RoleSpecifier::kPublic);
171  assert(&house_event != &HouseEvent::kTrue &&

181 void Serialize(const Model& model, std::FILE* out) {
182  xml::Stream xml_stream(out);
183  xml::StreamElement root = xml_stream.root("opsa-mef");
184  if (!model.HasDefaultName())
185  root.SetAttribute("name", model.name());
186  SerializeLabelAndAttributes(model, &root);
187  /// @todo Implement serialization for the following unsupported constructs.
188  assert(model.ccf_groups().empty());
189  assert(model.parameters().empty());
190  assert(model.initiating_events().empty());
191  assert(model.event_trees().empty());
192  assert(model.sequences().empty());
193  assert(model.rules().empty());
194 
195  for (const FaultTreePtr& fault_tree : model.fault_trees())
196  Serialize(*fault_tree, &root);
197 
198  xml::StreamElement model_data = root.AddChild("model-data");
199  for (const BasicEventPtr& basic_event : model.basic_events())
200  Serialize(*basic_event, &model_data);
201  for (const HouseEventPtr& house_event : model.house_events())
202  Serialize(*house_event, &model_data);
203 }
