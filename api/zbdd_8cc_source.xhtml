67 Zbdd::Zbdd(const Pdag* graph, const Settings& settings) noexcept
68  : Zbdd(graph->root(), settings) {
69  assert(!graph->complement() && "Complements must be propagated.");
70  if (graph->IsTrivial()) {
71  const Gate& top_gate = graph->root();
72  assert(top_gate.args().size() == 1);
73  assert(top_gate.args<Gate>().empty());
74  int child = *top_gate.args().begin();
75  if (top_gate.constant()) {
76  root_ = child < 0 ? kEmpty_ : kBase_;
77  } else if (child < 0) {
78  root_ = kBase_;
79  } else {
80  const Variable& var = top_gate.args<Variable>().begin()->second;
81  root_ = FindOrAddVertex(var.index(), kBase_, kEmpty_, var.order());
82  }
83  }
84  CHECK_ZBDD(true);
85 }

87 void Zbdd::Analyze(const Pdag* graph) noexcept {
88  CLOCK(zbdd_time);
89  assert(root_->terminal() ||
90  SetNode::Ref(root_).max_set_order() <= kSettings_.limit_order());
91  root_ = Minimize(root_); // Likely to be minimal by now.
92  assert(root_->terminal() || SetNode::Ref(root_).minimal());
93  for (const auto& entry : modules_)
94  entry.second->Analyze();
95 
96  Prune(root_, kSettings_.limit_order());
97  if (graph)
98  ApplySubstitutions(graph->substitutions());
99 
100  Freeze(); // Complete cleanup of the memory.
101  LOG(DEBUG3) << "G" << module_index_ << " analysis time: " << DUR(zbdd_time);
102 }

113 Zbdd::Zbdd(const Bdd::Function& module, bool coherent, Bdd* bdd,
114  const Settings& settings, int module_index) noexcept
115  : Zbdd(settings, coherent, module_index) {
116  CLOCK(init_time);
117  LOG(DEBUG2) << "Creating ZBDD from BDD: G" << module_index;
118  LOG(DEBUG4) << "Limit on product order: " << settings.limit_order();
119  PairTable<VertexPtr> ites;
120  root_ = Minimize(ConvertBdd(module.vertex, module.complement, bdd,
121  kSettings_.limit_order(), &ites));
122  assert(root_->terminal() || SetNode::Ref(root_).minimal());
123  Log();
124  LOG(DEBUG2) << "Created ZBDD from BDD in " << DUR(init_time);
125  std::map<int, std::pair<bool, int>> sub_modules;
126  GatherModules(root_, 0, &sub_modules);
127  for (const auto& entry : sub_modules) {
128  int index = entry.first;
129  assert(!modules_.count(index) && "Recalculating modules.");
130  Bdd::Function sub = bdd->modules().find(std::abs(index))->second;
131  assert(!sub.vertex->terminal() && "Unexpected BDD terminal vertex.");
132  int limit = entry.second.second;
133  assert(limit >= 0 && "Order cut-off is not strict.");
134  bool module_coherence = entry.second.first && (index > 0);
135  if (limit == 0 && module_coherence) { // Unity is impossible.
136  JoinModule(index, std::unique_ptr<Zbdd>(new Zbdd(settings)));
137  continue;
138  }
139  Settings adjusted(settings);
140  adjusted.limit_order(limit);
141  sub.complement ^= index < 0;
142  JoinModule(index, std::unique_ptr<Zbdd>(new Zbdd(sub, module_coherence, bdd,
143  adjusted, index)));
144  }
145  if (ext::any_of(modules_, [](const ModuleEntry& member) {
146  return member.second->root_->terminal();
147  })) {
148  LOG(DEBUG4) << "Eliminating constant modules from ZBDD...";
149  std::unordered_map<int, VertexPtr> results;
150  root_ = EliminateConstantModules(root_, &results);
151  }
152 }

154 Zbdd::Zbdd(const Gate& gate, const Settings& settings) noexcept
155  : Zbdd(settings, gate.coherent(), gate.index()) {
156  if (gate.constant() || gate.type() == kNull)
157  return;
158  assert(!settings.prime_implicants() && "Not implemented.");
159  CLOCK(init_time);
160  assert(gate.module() && "The constructor is meant for module gates.");
161  LOG(DEBUG3) << "Converting module to ZBDD: G" << gate.index();
162  LOG(DEBUG4) << "Limit on product order: " << settings.limit_order();
163  std::unordered_map<int, std::pair<VertexPtr, int>> gates;
164  std::unordered_map<int, const Gate*> module_gates;
165  root_ = ConvertGraph(gate, &gates, &module_gates);
166  if (!coherent_) {
167  LOG(DEBUG4) << "Eliminating complements from ZBDD...";
168  std::unordered_map<int, VertexPtr> results;
169  root_ = EliminateComplements(root_, &results);
170  }
171  LOG(DEBUG4) << "Minimizing ZBDD...";
172  root_ = Minimize(root_);
173  Log();
174  LOG(DEBUG3) << "Finished module conversion to ZBDD in " << DUR(init_time);
175  std::map<int, std::pair<bool, int>> sub_modules;
176  GatherModules(root_, 0, &sub_modules);
177  for (const auto& entry : sub_modules) {
178  int index = entry.first;
179  assert(index > 0 && "No complement gates.");
180  assert(!modules_.count(index) && "Recalculating modules.");
181  int limit = entry.second.second;
182  assert(limit >= 0 && "Order cut-off is not strict.");
183  bool coherent = entry.second.first;
184  if (limit == 0 && coherent) { // Unity is impossible.
185  JoinModule(index, std::unique_ptr<Zbdd>(new Zbdd(settings)));
186  continue;
187  }
188  const Gate* module_gate = module_gates.find(index)->second;
189  Settings adjusted(settings);
190  adjusted.limit_order(limit);
191  JoinModule(index, std::unique_ptr<Zbdd>(new Zbdd(*module_gate, adjusted)));
192  }
193  EliminateConstantModules();
194 }

198 SetNodePtr Zbdd::FindOrAddVertex(int index, const VertexPtr& high,
199  const VertexPtr& low, int order, bool module,
200  bool coherent) noexcept {
201  assert(high->id() != low->id() && "Reduction failure.");
202 
203  SetNodeWeakPtr& in_table =
204  unique_table_.FindOrAdd(index, high->id(), low->id());
205  if (!in_table.expired())
206  return in_table.lock();
207  assert(order > 0 && "Improper order.");
208  SetNodePtr node(new SetNode(index, order, set_id_++, high, low));
209  node->module(module);
210  node->coherent(coherent);
211  int high_order = high->terminal() ? 0 : SetNode::Ref(high).max_set_order();
212  high_order += !MayBeUnity(*node);
213  int low_order = low->terminal() ? 0 : SetNode::Ref(low).max_set_order();
214  node->max_set_order(std::max(high_order, low_order));
215 
216  in_table = node;
217  return node;
218 }

262 Zbdd::VertexPtr Zbdd::ConvertBdd(const Bdd::VertexPtr& vertex, bool complement,
263  Bdd* bdd_graph, int limit_order,
264  PairTable<VertexPtr>* ites) noexcept {
265  if (vertex->terminal())
266  return complement ? kEmpty_ : kBase_;
267  VertexPtr& result =
268  (*ites)[{complement ? -vertex->id() : vertex->id(), limit_order}];
269  if (result)
270  return result;
271  if (!coherent_ && kSettings_.prime_implicants()) {
272  result = ConvertBddPrimeImplicants(Ite::Ptr(vertex), complement, bdd_graph,
273  limit_order, ites);
274  } else {
275  result =
276  ConvertBdd(Ite::Ptr(vertex), complement, bdd_graph, limit_order, ites);
277  }
278  assert(result->terminal() ||
279  SetNode::Ref(result).max_set_order() <= limit_order);
280  return result;
281 }

283 Zbdd::VertexPtr Zbdd::ConvertBdd(const ItePtr& ite, bool complement,
284  Bdd* bdd_graph, int limit_order,
285  PairTable<VertexPtr>* ites) noexcept {
286  if (ite->module() && !ite->coherent())
287  return ConvertBddPrimeImplicants(ite, complement, bdd_graph, limit_order,
288  ites);
289  VertexPtr low = ConvertBdd(ite->low(), ite->complement_edge() ^ complement,
290  bdd_graph, limit_order, ites);
291  if (limit_order == 0) { // Cut-off on the set order.
292  if (low->terminal())
293  return low;
294  return kEmpty_;
295  }
296  VertexPtr high =
297  ConvertBdd(ite->high(), complement, bdd_graph, --limit_order, ites);
298  return GetReducedVertex(ite, false, high, low);
299 }

301 Zbdd::VertexPtr Zbdd::ConvertBddPrimeImplicants(
302  const ItePtr& ite, bool complement, Bdd* bdd_graph, int limit_order,
303  PairTable<VertexPtr>* ites) noexcept {
304  Bdd::Function common = Bdd::Consensus()(bdd_graph, ite, complement);
305  VertexPtr consensus = ConvertBdd(common.vertex, common.complement, bdd_graph,
306  limit_order, ites);
307  if (limit_order == 0) { // Cut-off on the product order.
308  if (consensus->terminal())
309  return consensus;
310  return kEmpty_;
311  }
312  int sublimit = limit_order - 1; // Assumes non-Unity element.
313  if (ite->module() && !kSettings_.prime_implicants()) {
314  assert(!ite->coherent() && "Only non-coherent modules through PI.");
315  sublimit += 1; // Unity modules may happen with minimal cut sets.
316  }
317  VertexPtr high =
318  ConvertBdd(ite->high(), complement, bdd_graph, sublimit, ites);
319  VertexPtr low = ConvertBdd(ite->low(), ite->complement_edge() ^ complement,
320  bdd_graph, sublimit, ites);
321  return GetReducedVertex(ite, false, high,
322  GetReducedVertex(ite, true, low, consensus));
323 }

325 Zbdd::VertexPtr Zbdd::ConvertGraph(
326  const Gate& gate, std::unordered_map<int, std::pair<VertexPtr, int>>* gates,
327  std::unordered_map<int, const Gate*>* module_gates) noexcept {
328  assert(!gate.constant() && "Unexpected constant gate!");
329  VertexPtr result;
330  if (auto it_entry = ext::find(*gates, gate.index())) {
331  std::pair<VertexPtr, int>& entry = it_entry->second;
332  result = entry.first;
333  assert(entry.second < gate.parents().size());
334  if (++entry.second == gate.parents().size())
335  gates->erase(it_entry);
336  return result;
337  }
338  std::vector<VertexPtr> args;
339  for (const Gate::ConstArg<Variable>& arg : gate.args<Variable>()) {
340  args.push_back(
341  FindOrAddVertex(arg.first, kBase_, kEmpty_, arg.second.order()));
342  }
343  for (const Gate::ConstArg<Gate>& arg : gate.args<Gate>()) {
344  assert(arg.first > 0 && "Complements must be pushed down to variables.");
345  if (arg.second.module()) {
346  module_gates->emplace(arg.first, &arg.second);
347  args.push_back(FindOrAddVertex(arg.second, kBase_, kEmpty_));
348  } else {
349  args.push_back(ConvertGraph(arg.second, gates, module_gates));
350  }
351  }
352  boost::sort(args, [](const VertexPtr& lhs, const VertexPtr& rhs) {
353  if (lhs->terminal())
354  return true;
355  if (rhs->terminal())
356  return false;
357  return SetNode::Ref(lhs).order() > SetNode::Ref(rhs).order();
358  });
359  auto it = args.cbegin();
360  for (result = *it++; it != args.cend(); ++it) {
361  result = Apply(gate.type(), result, *it, kSettings_.limit_order());
362  }
363  ClearTables();
364  assert(result);
365  assert(result->terminal() ||
366  SetNode::Ref(result).max_set_order() <= kSettings_.limit_order());
367  if (gate.parents().size() > 1)
368  gates->insert({gate.index(), {result, 1}});
369  return result;
370 }

395 /// Specialization of Apply for AND operator for non-terminal ZBDD vertices.
396 template <>
397 Zbdd::VertexPtr Zbdd::Apply<kAnd>(const SetNodePtr& arg_one,
398  const SetNodePtr& arg_two,
399  int limit_order) noexcept {
400  VertexPtr high;
401  VertexPtr low;
402  int limit_high = limit_order - !MayBeUnity(*arg_one);
403  if (arg_one->order() == arg_two->order() &&
404  arg_one->index() == arg_two->index()) { // The same variable.
405  // (x*f1 + f0) * (x*g1 + g0) = x*(f1*(g1 + g0) + f0*g1) + f0*g0
406  high = Apply<kOr>(
407  Apply<kAnd>(arg_one->high(),
408  Apply<kOr>(arg_two->high(), arg_two->low(), limit_high),
409  limit_high),
410  Apply<kAnd>(arg_one->low(), arg_two->high(), limit_high), limit_high);
411  low = Apply<kAnd>(arg_one->low(), arg_two->low(), limit_order);
412  } else {
413  assert((arg_one->order() < arg_two->order() ||
414  arg_one->index() > arg_two->index()) &&
415  "Ordering contract failed.");
416  if (arg_one->order() == arg_two->order()) {
417  // (x*f1 + f0) * (~x*g1 + g0) = x*f1*g0 + f0*(~x*g1 + g0)
418  high = Apply<kAnd>(arg_one->high(), arg_two->low(), limit_high);
419  } else {
420  high = Apply<kAnd>(arg_one->high(), arg_two, limit_high);
421  }
422  low = Apply<kAnd>(arg_one->low(), arg_two, limit_order);
423  }
424  if (!high->terminal() && SetNode::Ref(high).order() == arg_one->order()) {
425  assert(SetNode::Ref(high).index() < arg_one->index());
426  high = SetNode::Ref(high).low();
427  }
428  return Minimize(GetReducedVertex(arg_one, high, low));
429 }

431 /// Specialization of Apply for AND operator for any ZBDD vertices.
432 template <>
433 Zbdd::VertexPtr Zbdd::Apply<kAnd>(const VertexPtr& arg_one,
434  const VertexPtr& arg_two,
435  int limit_order) noexcept {
436  if (limit_order < 0)
437  return kEmpty_;
438  if (arg_one->terminal()) {
439  if (Terminal<SetNode>::Ref(arg_one).value())
440  return Prune(arg_two, limit_order);
441  return kEmpty_;
442  }
443  if (arg_two->terminal()) {
444  if (Terminal<SetNode>::Ref(arg_two).value())
445  return Prune(arg_one, limit_order);
446  return kEmpty_;
447  }
448  if (arg_one->id() == arg_two->id())
449  return Prune(arg_one, limit_order);
450 
451  VertexPtr& result = and_table_[GetResultKey(arg_one, arg_two, limit_order)];
452  if (result)
453  return result; // Already computed.
454 
455  SetNodePtr set_one = SetNode::Ptr(arg_one);
456  SetNodePtr set_two = SetNode::Ptr(arg_two);
457  if (set_one->order() > set_two->order()) {
458  std::swap(set_one, set_two);
459  } else if (set_one->order() == set_two->order() &&
460  set_one->index() < set_two->index()) {
461  std::swap(set_one, set_two);
462  }
463  result = Apply<kAnd>(set_one, set_two, limit_order);
464  assert(result->terminal() ||
465  SetNode::Ref(result).max_set_order() <= limit_order);
466  return result;
467 }

471 Zbdd::VertexPtr Zbdd::Apply<kOr>(const SetNodePtr& arg_one,
472  const SetNodePtr& arg_two,

546 Zbdd::VertexPtr Zbdd::EliminateComplements(
547  const VertexPtr& vertex,
548  std::unordered_map<int, VertexPtr>* wide_results) noexcept {

635 Zbdd::VertexPtr Zbdd::Subsume(const VertexPtr& high,
636  const VertexPtr& low) noexcept {
637  if (low->terminal())
638  return Terminal<SetNode>::Ref(low).value() ? kEmpty_ : high;
639  if (high->terminal())
640  return high; // No need to reduce terminal sets.
641  VertexPtr& computed = subsume_table_[{high->id(), low->id()}];
642  if (computed)
643  return computed;
644 
645  SetNodePtr high_node = SetNode::Ptr(high);
646  SetNodePtr low_node = SetNode::Ptr(low);
647  if (high_node->order() > low_node->order() ||
648  (high_node->order() == low_node->order() &&
649  high_node->index() < low_node->index())) {
650  computed = Subsume(high, low_node->low());
651  return computed;
652  }
653  VertexPtr subhigh;
654  VertexPtr sublow;
655  if (high_node->order() == low_node->order() &&
656  high_node->index() == low_node->index()) {
657  assert(high_node->index() == low_node->index());
658  subhigh = Subsume(high_node->high(), low_node->high());
659  subhigh = Subsume(subhigh, low_node->low());
660  sublow = Subsume(high_node->low(), low_node->low());
661  } else {
662  assert(high_node->order() < low_node->order() ||
663  (high_node->order() == low_node->order() &&
664  high_node->index() > low_node->index()));
665  subhigh = Subsume(high_node->high(), low);
666  sublow = Subsume(high_node->low(), low);
667  }
668  if (subhigh->terminal() && !Terminal<SetNode>::Ref(subhigh).value()) {
669  computed = sublow;
670  return computed;
671  }
672  assert(subhigh->id() != sublow->id());
673  SetNodePtr new_high = FindOrAddVertex(high_node, subhigh, sublow);
674  new_high->minimal(high_node->minimal());
675  computed = new_high;
676  return computed;
677 }

701 bool Zbdd::MayBeUnity(const SetNode& node) noexcept {
702  if (kSettings_.prime_implicants())
703  return false;
704  // Unity node tests for minimal cut sets.
705  if (node.index() < 0)
706  return true; // Unity complement vars.
707  // Non-modular gates can be implied by other gates in the product;
708  // that is, (G1 & G2 = I & G2) if G2 implies G1.
709  //
710  // Non-coherent gates contain complements to be approximated to Unity.
711  if (this->IsGate(node))
712  return !node.module() || !node.coherent();
713  return false; // Positive non-gate variable.
714 }

716 int Zbdd::GatherModules(const VertexPtr& vertex, int current_order,
717  std::map<int, std::pair<bool, int>>* modules) noexcept {
718  assert(current_order >= 0);
719  if (vertex->terminal())
720  return Terminal<SetNode>::Ref(vertex).value() ? 0 : -1;
721  SetNode& node = SetNode::Ref(vertex);
722  int contribution = !MayBeUnity(node);
723  int high_order = current_order + contribution;
724  int min_high = GatherModules(node.high(), high_order, modules);
725  assert(min_high >= 0 && "No terminal Empty should be on high branch.");
726  if (node.module()) {
727  int module_order = kSettings_.limit_order() - min_high - current_order;
728  assert(module_order >= 0 && "Improper application of a cut-off.");
729  if (auto it = ext::find(*modules, node.index())) {
730  std::pair<bool, int>& entry = it->second;
731  assert(entry.first == node.coherent() && "Inconsistent flags.");
732  entry.second = std::max(entry.second, module_order);
733  } else {
734  modules->insert({node.index(), {node.coherent(), module_order}});
735  }
736  }
737  int min_low = GatherModules(node.low(), current_order, modules);
738  assert(min_low >= -1);
739  if (min_low == -1)
740  return min_high + contribution;
741  return std::min(min_high + contribution, min_low);
742 }

744 void Zbdd::ApplySubstitutions(
745  const std::vector<Pdag::Substitution>& substitutions) noexcept {
746  if (substitutions.empty())
747  return;
748  ClearTables();
749  unique_table_.clear(); // New ordering for nodes.
750 
751  VertexPtr new_root = kEmpty_;
752  for (const std::vector<int>& product : *this) {
753  std::vector<int> to_add;
754  std::vector<int> to_remove;
755  for (const Pdag::Substitution& substitution : substitutions) {
756  if (ext::all_of(substitution.hypothesis, [&product](int id) {
757  return boost::find(product, id) != product.end();
758  })) {
759  to_remove.insert(to_remove.end(), substitution.source.begin(),
760  substitution.source.end());
761  if (substitution.target)
762  to_add.push_back(substitution.target);
763  }
764  }
765  VertexPtr new_product = kBase_;
766  for (int id : product) {
767  if (boost::find(to_remove, id) != to_remove.end())
768  continue;
769  new_product = Apply<kAnd>(
770  new_product, FindOrAddVertex(id, kBase_, kEmpty_, std::abs(id)),
771  kSettings_.limit_order());
772  }
773  for (int id : to_add) {
774  new_product = Apply<kAnd>(
775  new_product, FindOrAddVertex(id, kBase_, kEmpty_, std::abs(id)),
776  kSettings_.limit_order());
777  }
778  new_root = Apply<kOr>(new_root, new_product, kSettings_.limit_order());
779  }
780  root_ = std::move(new_root);
781  root_ = Minimize(root_);
782 }
