32 void Element::name(std::string name) {
33  if (name.empty())
34  SCRAM_THROW(LogicError("The element name cannot be empty"));
35  if (name.find('.') != std::string::npos)
36  SCRAM_THROW(ValidityError("The element name is malformed."));
37  name_ = std::move(name);
38 }

49 void Element::SetAttribute(Attribute attr) {
50  auto it = boost::find_if(attributes_, [&attr](const Attribute& member) {
51  return attr.name == member.name;
52  });
53  if (it != attributes_.end()) {
54  it->value = std::move(attr.value);
55  it->type = std::move(attr.type);
56  } else {
57  attributes_.emplace_back(std::move(attr));

67 const Attribute& Element::GetAttribute(const std::string& name) const {
68  auto it = boost::find_if(attributes_, [&name](const Attribute& attr) {
69  return attr.name == name;
70  });
71  if (it == attributes_.end())
72  SCRAM_THROW(LogicError("Element does not have attribute: " + name));

77 bool Element::RemoveAttribute(const std::string& name) {
78  auto it = boost::find_if(attributes_, [&name](const Attribute& attr) {
79  return attr.name == name;
80  });
81  if (it == attributes_.end())
82  return false;
83  attributes_.erase(it);
84  return true;
