60 /// Abstract base class for all sorts of expressions to describe events.
61 /// This class also acts like a connector for parameter nodes
62 /// and may create cycles.
63 /// Expressions are not expected to be shared
64 /// except for parameters.
65 /// In addition, expressions are not expected to be changed
66 /// after validation phases.
67 class Expression : private boost::noncopyable {
68  public:
69  /// Constructor for use by derived classes
70  /// to register their arguments.
71  ///
72  /// @param[in] args Arguments of this expression.
73  explicit Expression(std::vector<Expression*> args = {});
74 
75  virtual ~Expression() = default;
76 
77  /// @returns A set of arguments of the expression.
78  const std::vector<Expression*>& args() const { return args_; }
79 
80  /// Validates the expression.
81  /// This late validation is due to parameters that are defined late.
82  ///
83  /// @throws DomainError The argument value domains are invalid.
84  /// @throws ValidityError The arguments are invalid for setup.
85  virtual void Validate() const {}
86 
87  /// @returns The mean value of this expression.
88  virtual double value() noexcept = 0;
89 
90  /// @returns The domain interval for validation purposes only.
91  virtual Interval interval() noexcept {
92  double value = this->value();
93  return Interval::closed(value, value);
94  }
95 
96  /// Determines if the value of the expression contains deviate expressions.
97  /// The default logic is to check arguments with uncertainties for sampling.
98  /// Derived expression classes must decide
99  /// if they don't have arguments,
100  /// or if they are random deviates.
101  ///
102  /// @returns true if the expression's value deviates from its mean.
103  /// @returns false if the expression's value does not need sampling.
104  ///
105  /// @warning Improper registration of arguments
106  /// may yield silent failure.
107  virtual bool IsDeviate() noexcept;
108 
109  /// @returns A sampled value of this expression.
110  double Sample() noexcept;
111 
112  /// This routine resets the sampling to get new values.
113  /// All the arguments are called to reset themselves.
114  /// If this expression was not sampled,
115  /// its arguments are not going to get any calls.
116  void Reset() noexcept;
117 
118  protected:
119  /// Registers an additional argument expression.
120  ///
121  /// @param[in] arg An argument expression used by this expression.
122  void AddArg(Expression* arg) { args_.push_back(arg); }
123 
124  private:
125  /// Runs sampling of the expression.
126  /// Derived concrete classes must provide the calculation.
127  ///
128  /// @returns A sampled value of this expression.
129  virtual double DoSample() noexcept = 0;
130 
131  std::vector<Expression*> args_; ///< Expression's arguments.
132  double sampled_value_; ///< The sampled value.
133  bool sampled_; ///< Indication if the expression is already sampled.
134 };

165 /// Unary expression.
166 template <typename T>
167 class NaryExpression<T, 1> : public ExpressionFormula<NaryExpression<T, 1>> {
168  public:
169  /// @param[in] expression The single argument.
170  explicit NaryExpression(Expression* expression)
171  : ExpressionFormula<NaryExpression<T, 1>>({expression}),
172  expression_(*expression) {}
173 
174  void Validate() const override {}
175 
176  Interval interval() noexcept override {
177  Interval arg_interval = expression_.interval();
178  double max_value = T()(arg_interval.upper());
179  double min_value = T()(arg_interval.lower());
180  auto min_max = std::minmax(max_value, min_value);
181  return Interval::closed(min_max.first, min_max.second);
182  }

196 class NaryExpression<T, 2> : public ExpressionFormula<NaryExpression<T, 2>> {
197  public:

202  void Validate() const override {}

204  Interval interval() noexcept override {
205  Interval interval_one = Expression::args().front()->interval();
206  Interval interval_two = Expression::args().back()->interval();
207  double max_max = T()(interval_one.upper(), interval_two.upper());
208  double max_min = T()(interval_one.upper(), interval_two.lower());
209  double min_max = T()(interval_one.lower(), interval_two.upper());
210  double min_min = T()(interval_one.lower(), interval_two.lower());
211  auto interval_pair = std::minmax({max_max, max_min, min_max, min_min});
212  return Interval::closed(interval_pair.first, interval_pair.second);
213  }

234 /// Multivariate expression.
235 template <typename T>
236 class NaryExpression<T, -1> : public ExpressionFormula<NaryExpression<T, -1>> {
237  public:
238  /// Checks the number of provided arguments upon initialization.
239  ///
240  /// @param[in] args Arguments of this expression.
241  ///
242  /// @throws ValidityError The number of arguments is fewer than 2.
243  explicit NaryExpression(std::vector<Expression*> args)
244  : ExpressionFormula<NaryExpression<T, -1>>(std::move(args)) {
245  detail::EnsureMultivariateArgs(Expression::args());
246  }
247 
248  void Validate() const override {}
249 
250  Interval interval() noexcept override {
251  auto it = Expression::args().begin();
252  Interval first_arg_interval = (*it)->interval();
253  double max_value = first_arg_interval.upper();
254  double min_value = first_arg_interval.lower();
255  for (++it; it != Expression::args().end(); ++it) {
256  Interval next_arg_interval = (*it)->interval();
257  double arg_max = next_arg_interval.upper();
258  double arg_min = next_arg_interval.lower();
259  double max_max = T()(max_value, arg_max);
260  double max_min = T()(max_value, arg_min);
261  double min_max = T()(min_value, arg_max);
262  double min_min = T()(min_value, arg_min);
263  std::tie(min_value, max_value) =
264  std::minmax({max_max, max_min, min_max, min_min});
265  }
266  assert(min_value <= max_value);
267  return Interval::closed(min_value, max_value);
268  }

270  /// Computes the expression value with a given argument value extractor.
271  template <typename F>
272  double Compute(F&& eval) noexcept {
273  auto it = Expression::args().begin();
274  double result = eval(*it);
275  for (++it; it != Expression::args().end(); ++it) {
276  result = T()(result, eval(*it));
277  }
278  return result;
279  }
