119 /// Uncertainty analysis facility.

141 template <class Calculator>
142 std::vector<double> UncertaintyAnalyzer<Calculator>::Sample() noexcept {
143  std::vector<std::pair<int, mef::Expression&>> deviate_expressions =
144  UncertaintyAnalysis::GatherDeviateExpressions(prob_analyzer_->graph());
145  Pdag::IndexMap<double> p_vars = prob_analyzer_->p_vars(); // Private copy!
146  std::vector<double> samples;
147  samples.reserve(Analysis::settings().num_trials());
148 
149  for (int i = 0; i < Analysis::settings().num_trials(); ++i) {
150  UncertaintyAnalysis::SampleExpressions(deviate_expressions, &p_vars);
151  double result = prob_analyzer_->CalculateTotalProbability(p_vars);
152  assert(result >= 0 && result <= 1);
153  samples.push_back(result);
154  }
155 
156  return samples;
157 }
