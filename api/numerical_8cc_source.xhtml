40 void Mod::Validate() const {
41  assert(args().size() == 2);
42  auto* arg_two = args().back();
43  int arg_value = arg_two->value();
44  if (arg_value == 0)
45  SCRAM_THROW(DomainError("Modulo second operand must not be 0."));
46  Interval interval = arg_two->interval();
47  int high = interval.upper();
48  int low = interval.lower();
49  if (high == 0 || low == 0 || (low < 0 && 0 < high)) {
50  SCRAM_THROW(
51  DomainError("Modulo second operand sample must not contain 0."));
52  }

59  auto* arg_two = args().back();

61  SCRAM_THROW(DomainError("0 to power 0 or less is undefined."));

63  SCRAM_THROW(
