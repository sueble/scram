44 std::string normalize(const std::string& file_path, const fs::path& base_path) {
45  fs::path abs_path = fs::absolute(file_path, base_path).generic_string();
46 #if BOOST_OS_WINDOWS
47  return abs_path.generic_string();
48 #else
49  std::string abnormal_path = abs_path.string();

57 Config::Config(const std::string& config_file) {
58  static xml::Validator validator(env::config_schema());
59 
60  if (fs::exists(config_file) == false) {
61  SCRAM_THROW(IOError("The configuration file does not exist."))
62  << boost::errinfo_file_name(config_file);
63  }
64 
65  xml::Document document(config_file, &validator);
66  xml::Element root = document.root();
67  assert(root.name() == "scram");
68  fs::path base_path = fs::path(config_file).parent_path();
69  GatherInputFiles(root, base_path);
70 
71  if (std::optional<xml::Element> out = root.child("output-path")) {
72  output_path_ = normalize(std::string(out->text()), base_path);
73  }
74 
75  try {
76  GatherOptions(root);
77  } catch (Error& err) {
78  err << boost::errinfo_file_name(config_file);
79  throw;
80  }
81 }

83 void Config::GatherInputFiles(const xml::Element& root,
84  const fs::path& base_path) {
85  std::optional<xml::Element> input_files = root.child("input-files");
86  if (!input_files)
87  return;
88  for (xml::Element input_file : input_files->children()) {
89  assert(input_file.name() == "file");
90  input_files_.push_back(
91  normalize(std::string(input_file.text()), base_path));
92  }

95 void Config::GatherOptions(const xml::Element& root) {
96  std::optional<xml::Element> options_element = root.child("options");
97  if (!options_element)
98  return;
99  // The loop is used instead of query
100  // because the order of options matters,
101  // yet this function should not know what the order is.
102  for (xml::Element option_group : options_element->children()) {
103  try {
104  std::string_view name = option_group.name();
105  if (name == "algorithm") {
106  settings_.algorithm(option_group.attribute("name"));
107 
108  } else if (name == "prime-implicants") {
109  settings_.prime_implicants(true);
110 
111  } else if (name == "approximation") {
112  settings_.approximation(option_group.attribute("name"));
113 
114  } else if (name == "limits") {
115  SetLimits(option_group);
116  }
117  } catch (SettingsError& err) {
118  err << boost::errinfo_at_line(option_group.line());
119  throw;
120  }
121  }
122  if (std::optional<xml::Element> analysis_group =
123  options_element->child("analysis")) {
124  try {
125  SetAnalysis(*analysis_group);
126  } catch (SettingsError& err) {
127  err << boost::errinfo_at_line(analysis_group->line());
128  throw;
129  }
130  }
131 }

135  if (std::optional<bool> flag = analysis.attribute<bool>(tag))
136  setter(*flag);

138  set_flag("probability",

144  set_flag("ccf", [this](bool flag) { settings_.ccf_analysis(flag); });
