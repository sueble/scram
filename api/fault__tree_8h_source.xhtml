35 /// Component is for logical grouping of events, gates, and other components.
36 class Component : public Element, public Role {
37  public:
38  /// Constructs a component assuming
39  /// that it exists within some fault tree.
40  /// The public or private role of a component is not
41  /// for the components itself,
42  /// but for the events and parameters of the component.
43  /// Component name is not meant to be public;
44  /// however, it must be unique within the parent fault tree or component.
45  ///
46  /// @param[in] name The name identifier for the component.
47  /// @param[in] base_path The series of containers to get this container.
48  /// @param[in] role The default role for container members.
49  ///
50  /// @throws LogicError The name is empty.
51  /// @throws ValidityError The name or reference paths are malformed.
52  explicit Component(std::string name, std::string base_path = "",
53  RoleSpecifier role = RoleSpecifier::kPublic);

58  /// with construct original names as keys.

135 /// Fault tree representation as a container of
136 /// gates, basic and house events, and other information.
137 /// Additional functionality of a fault tree includes
138 /// detection of top events.
139 class FaultTree : public Component {
140  public:
141  /// The main constructor of the Fault Tree.
142  /// Fault trees are assumed to be public and belong to the root model.
143  ///
144  /// @param[in] name The name identifier of this fault tree.
145  explicit FaultTree(const std::string& name);
146 
147  /// @returns The collected top events of this fault tree.
148  const std::vector<const Gate*>& top_events() const { return top_events_; }
149 
150  /// Collects top event gates in this fault tree with components.
151  /// This function is essential to guess the analysis targets
152  /// if the user does not supply any.
153  /// If the structure of the fault tree changes,
154  /// this function must be called again to update the top events.
155  ///
156  /// @pre Gate marks are clear.
157  void CollectTopEvents();
158 
159  private:
160  /// Recursively marks descendant gates
161  /// with an unspecified-but-non-clear mark.
162  /// These gates belong to this fault tree only.
163  ///
164  /// @param[in,out] gate The ancestor gate.
165  /// @param[in] gates Gates belonging to the whole fault tree with components.
166  void MarkNonTopGates(Gate* gate, const std::unordered_set<Gate*>& gates);
167 
168  /// Recursively marks descendant gates in formulas as "non-top"
169  ///
170  /// @param[in] formula The formula of a gate or another formula.
171  /// @param[in] gates Gates belonging to the whole fault tree with components.
172  void MarkNonTopGates(const Formula& formula,
173  const std::unordered_set<Gate*>& gates);
174 
175  std::vector<const Gate*> top_events_; ///< Top events of this fault tree.
176 };
