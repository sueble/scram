107 /// Parses the command-line arguments.
108 ///
109 /// @param[in] argc Count of arguments.
110 /// @param[in] argv Values of arguments.
111 /// @param[out] vm Variables map of program options.
112 ///
113 /// @returns 0 for success.
114 /// @returns 1 for errored state.
115 /// @returns -1 for information only state like help and version.
116 int ParseArguments(int argc, char* argv[], po::variables_map* vm) {
117  const char* usage = "Usage: scram [options] input-files...";
118  po::options_description desc = ConstructOptions();
119  try {
120  po::store(po::parse_command_line(argc, argv, desc), *vm);
121  } catch (std::exception& err) {
122  std::cerr << "Option error: " << err.what() << "\n\n"
123  << usage << "\n\n"
124  << desc << std::endl;
125  return 1;
126  }
127  po::options_description options("All options with positional input files.");
128  options.add(desc).add_options()("input-files",
129  po::value<std::vector<std::string>>(),
130  "XML input files with analysis constructs");
131  po::positional_options_description p;
132  p.add("input-files", -1); // All input files are implicit.
133  po::store(
134  po::command_line_parser(argc, argv).options(options).positional(p).run(),
135  *vm);
136  po::notify(*vm);
137 
138  auto print_help = [&usage, &desc](std::ostream& out) {
139  out << usage << "\n\n" << desc << std::endl;
140  };
141 
142  // Process command-line arguments.
143  if (vm->count("help")) {
144  print_help(std::cout);
145  return -1;
146  }
147  if (vm->count("version")) {
148  std::cout << "SCRAM " << scram::version::core() << " ("
149  << scram::version::describe() << ")"
150  << "\n\nDependencies:\n"
151  << " Boost " << scram::version::boost() << "\n"
152  << " libxml2 " << scram::version::libxml() << std::endl;
153  return -1;
154  }
155 
156  if (vm->count("verbosity")) {
157  int level = (*vm)["verbosity"].as<int>();
158  if (level < 0 || level > scram::kMaxVerbosity) {
159  std::cerr << "Log verbosity must be between 0 and "
160  << scram::kMaxVerbosity << ".\n\n";
161  print_help(std::cerr);
162  return 1;
163  }
164  }
165 
166  if (!vm->count("input-files") && !vm->count("config-file")) {
167  std::cerr << "No input or configuration file is given.\n\n";
168  print_help(std::cerr);
169  return 1;
170  }
171  if ((vm->count("bdd") + vm->count("zbdd") + vm->count("mocus")) > 1) {
172  std::cerr << "Mutually exclusive qualitative analysis algorithms.\n"
173  << "(MOCUS/BDD/ZBDD) cannot be applied at the same time.\n\n";
174  print_help(std::cerr);
175  return 1;
176  }
177  if (vm->count("rare-event") && vm->count("mcub")) {
178  std::cerr << "The rare event and MCUB approximations cannot be "
179  << "applied at the same time.\n\n";
180  print_help(std::cerr);
181  return 1;
182  }
183  return 0;
184 }

194 /// Updates analysis settings from command-line arguments.
195 ///
196 /// @param[in] vm Variables map of program options.
197 /// @param[in,out] settings Pre-configured or default settings.
198 ///
199 /// @throws SettingsError The indication of an error in arguments.
200 /// @throws std::exception vm does not contain a required option.
201 /// At least defaults are expected.
202 void ConstructSettings(const po::variables_map& vm,
203  scram::core::Settings* settings) {
204  if (vm.count("bdd")) {
205  settings->algorithm("bdd");
206  } else if (vm.count("zbdd")) {
207  settings->algorithm("zbdd");
208  } else if (vm.count("mocus")) {
209  settings->algorithm("mocus");
210  }
211  settings->prime_implicants(vm.count("prime-implicants"));
212  // Determine if the probability approximation is requested.
213  if (vm.count("rare-event")) {
214  assert(!vm.count("mcub"));
215  settings->approximation("rare-event");
216  } else if (vm.count("mcub")) {
217  settings->approximation("mcub");
218  }
219  SET("time-step", double, time_step);
220  SET("sil", bool, safety_integrity_levels);
221 
222  SET("probability", bool, probability_analysis);
223  SET("importance", bool, importance_analysis);
224  SET("uncertainty", bool, uncertainty_analysis);
225  SET("ccf", bool, ccf_analysis);
226  SET("seed", int, seed);
227  SET("limit-order", int, limit_order);
228  SET("cut-off", double, cut_off);
229  SET("mission-time", double, mission_time);
230  SET("num-trials", int, num_trials);
231  SET("num-quantiles", int, num_quantiles);
232  SET("num-bins", int, num_bins);
233 #ifndef NDEBUG
234  settings->preprocessor = vm.count("preprocessor");
235  settings->print = vm.count("print");
236 #endif
237 }

240 /// Main body of command-line entrance to run the program.
241 ///
242 /// @param[in] vm Variables map of program options.
243 ///
244 /// @throws Error Exceptions specific to SCRAM.
245 /// @throws boost::exception Boost errors with the variables map.
246 /// @throws std::exception All other problems.
247 void RunScram(const po::variables_map& vm) {
248  scram::core::Settings settings; // Analysis settings.
249  std::vector<std::string> input_files;
250  std::string output_path;
251  // Get configurations if any.
252  // Invalid configurations will throw.
253  if (vm.count("config-file")) {
254  auto config =
255  std::make_unique<scram::Config>(vm["config-file"].as<std::string>());
256  settings = config->settings();
257  input_files = config->input_files();
258  output_path = config->output_path();
259  }
260  // Command-line settings overwrite
261  // the settings from the configurations.
262  ConstructSettings(vm, &settings);
263  if (vm.count("input-files")) {
264  auto cmd_input = vm["input-files"].as<std::vector<std::string>>();
265  input_files.insert(input_files.end(), cmd_input.begin(), cmd_input.end());
266  }
267  if (vm.count("output-path")) {
268  output_path = vm["output-path"].as<std::string>();
269  }
270  // Process input files
271  // into valid analysis containers and constructs.
272  // Throws if anything is invalid.
273  std::shared_ptr<scram::mef::Model> model =
274  scram::mef::Initializer(input_files, settings, vm.count("allow-extern"))
275  .model();
276 #ifndef NDEBUG
277  if (vm.count("serialize"))
278  return Serialize(*model, stdout);
279 #endif
280  if (vm.count("validate"))
281  return; // Stop if only validation is requested.
282 
283  // Initiate risk analysis with the given information.
284  scram::core::RiskAnalysis analysis(model.get(), settings);
285  analysis.Analyze();
286 #ifndef NDEBUG
287  if (vm.count("no-report") || vm.count("preprocessor") || vm.count("print"))
288  return;
289 #endif
290  scram::Reporter reporter;
291  bool indent = vm.count("no-indent") ? false : true;
292  if (output_path.empty()) {
293  reporter.Report(analysis, stdout, indent);
294  } else {
295  reporter.Report(analysis, output_path, indent);
296  }
297 }

299 /// Callback function to redirect XML library error/warning messages to logging.
300 /// Otherwise, the messages are printed to the standard error.
301 ///
302 /// @param[in] msg The printf-style format string.
303 /// @param[in] ... The variadic arguments for the format string.
304 ///
305 /// @pre The library strictly follows validity conditions of printf.
306 void LogXmlError(void* /*ctx*/, const char* msg, ...) noexcept {
307  std::va_list args;
308  va_start(args, msg);
309  SCOPE_EXIT([&args] { va_end(args); });
310 
311  std::va_list args_for_nchar; // Only used to determine the string length.
312  va_copy(args_for_nchar, args);
313  int nchar = std::vsnprintf(nullptr, 0, msg, args_for_nchar);
314  va_end(args_for_nchar);
315  if (nchar < 0) {
316  LOG(scram::ERROR) << "String formatting failure: " << std::strerror(errno);
317  return;
318  }
319 
320  std::vector<char> buffer(nchar + /*null terminator*/ 1);
321  std::vsnprintf(buffer.data(), buffer.size(), msg, args);
322  LOG(scram::WARNING) << buffer.data();
323 }

327 /// Command-line SCRAM entrance.
328 ///
329 /// @param[in] argc Argument count.
330 /// @param[in] argv Argument vector.
331 ///
332 /// @returns 0 for success.
333 /// @returns 1 for errored state.
334 int main(int argc, char* argv[]) {
335  LIBXML_TEST_VERSION
336  xmlInitParser();
337  SCOPE_EXIT(&xmlCleanupParser);
338 
339  xmlGenericErrorFunc xml_error_printer = LogXmlError;
340  initGenericErrorDefaultFunc(&xml_error_printer);
341 
342  try {
343  // Parse command-line options.
344  po::variables_map vm;
345  int ret = ParseArguments(argc, argv, &vm);
346  if (ret == 1)
347  return 1;
348 
349  if (vm.count("verbosity")) {
350  scram::Logger::report_level(
351  static_cast<scram::LogLevel>(vm["verbosity"].as<int>()));
352  }
353 
354  if (ret == 0)
355  RunScram(vm);
356  } catch (const scram::LogicError& err) {
357  LOG(scram::ERROR) << "Logic Error:\n" << boost::diagnostic_information(err);
358  return 1;
359  } catch (const scram::IOError& err) {
360  LOG(scram::DEBUG1) << boost::diagnostic_information(err);
361  std::cerr << boost::core::demangled_name(typeid(err)) << "\n\n";
362  const std::string* filename =
363  boost::get_error_info<boost::errinfo_file_name>(err);
364  assert(filename);
365  std::cerr << "File: " << *filename << "\n";
366  if (const std::string* mode =
367  boost::get_error_info<boost::errinfo_file_open_mode>(err)) {
368  std::cerr << "Open mode: " << *mode << "\n";
369  }
370 
371  if (const int* errnum = boost::get_error_info<boost::errinfo_errno>(err)) {
372  std::cerr << "Error code: " << *errnum << "\n";
373  std::cerr << "Error string: " << std::strerror(*errnum) << "\n";
374  }
375  std::cerr << "\n" << err.what() << std::endl;
376  return 1;
377  } catch (const scram::Error& err) {
378  LOG(scram::DEBUG1) << boost::diagnostic_information(err);
379  std::cerr << boost::core::demangled_name(typeid(err)) << "\n\n";
380  if (const std::string* filename =
381  boost::get_error_info<boost::errinfo_file_name>(err)) {
382  std::cerr << "File: " << *filename << "\n";
383  if (const int* line = boost::get_error_info<boost::errinfo_at_line>(err))
384  std::cerr << "Line: " << *line << "\n";
385  }
386  if (const std::string* container =
387  boost::get_error_info<scram::mef::errinfo_container>(err)) {
388  std::cerr << "MEF Container: " << *container << "\n";
389  }
390  if (const std::string* xml_element =
391  boost::get_error_info<scram::xml::errinfo_element>(err)) {
392  std::cerr << "XML element: " << *xml_element << "\n";
393  }
394  if (const std::string* xml_attribute =
395  boost::get_error_info<scram::xml::errinfo_attribute>(err)) {
396  std::cerr << "XML attribute: " << *xml_attribute << "\n";
397  }
398  std::cerr << "\n" << err.what() << std::endl;
399  return 1;
400  } catch (const boost::exception& boost_err) {
401  LOG(scram::ERROR) << "Unexpected Boost Exception:\n"
402  << boost::diagnostic_information(boost_err);
403  return 1;
404  } catch (const std::exception& err) {
405  LOG(scram::ERROR) << "Unexpected Exception: "
406  << boost::core::demangled_name(typeid(err)) << ": "
407  << err.what();
408  return 1;
409  }
